Foundation

What is Sorting?

Sorting is the process of arranging or ordering a collection of items such that each item and its successor satisfy a prescribed relationship.

The items can be simple values — such as integers and reals — or more complex types such as student records or dictionary entries. In either case, the ordering of the items is based on the value of a sort key.

The key is the value itself when sorting simple types, or it can be a specific component (or combination of components) when sorting complex types. We encounter sorting in everyday life: phone books, dictionaries, and indexes are all organized alphabetically to make finding entries much easier.

Bubble Sort

Compare & Swap Adjacent Elements
Rearranges values by iterating over the list multiple times, causing larger values to "bubble" to the top (or end) of the list. In each pass, adjacent elements are compared and swapped if they are out of order.

🐍 Python Implementation
def bubble_sort(arr):
    # Get the number of elements
    n = len(arr)

    for i in range(n):
        swapped = False  # Optimization flag

        # Each pass places the next largest at position n-i-1
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                # Swap adjacent elements
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True

        # Early exit: already sorted
        if not swapped:
            break

    return arr


# ── Example Usage ──────────────────────────────────────
nums = [64, 34, 25, 12, 22, 11, 90]
result = bubble_sort(nums.copy())

print("Original:", nums)    # [64, 34, 25, 12, 22, 11, 90]
print("Sorted:  ", result)  # [11, 12, 22, 25, 34, 64, 90]


# ── Step-by-step trace ─────────────────────────────────
def bubble_sort_verbose(arr):
    n = len(arr)
    for i in range(n):
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
        print(f"Pass {i+1}: {arr}")
    return arr

Selection Sort

Find Minimum, Place at Front
A simple sorting algorithm that sorts an array by repeatedly finding the minimum element from the unsorted portion and placing it at the beginning. With each pass, the sorted portion grows by one element.
⚡ Live Visualizer — Selection Sort
Press ▶ Start to begin
🐍 Python Implementation
def selection_sort(arr):
    n = len(arr)

    for i in range(n):
        # Assume current index holds the minimum
        min_idx = i

        # Scan unsorted portion for actual minimum
        for j in range(i + 1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j

        # Swap the found minimum with position i
        arr[i], arr[min_idx] = arr[min_idx], arr[i]

    return arr


# ── Example Usage ──────────────────────────────────────
nums = [64, 25, 12, 22, 11]
result = selection_sort(nums.copy())

print("Original:", nums)    # [64, 25, 12, 22, 11]
print("Sorted:  ", result)  # [11, 12, 22, 25, 64]


# ── Trace: see which elements get swapped ──────────────
def selection_sort_verbose(arr):
    n = len(arr)
    for i in range(n):
        min_idx = i
        for j in range(i + 1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        print(f"Swap arr[{i}]={arr[i]} ↔ arr[{min_idx}]={arr[min_idx]}")
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
        print(f"  After: {arr}")
    return arr

Merge Sort

Divide & Conquer Recursion
A divide-and-conquer algorithm that recursively divides the list into smaller sublists, sorts each half, and then merges them back together in sorted order. It guarantees O(n log n) performance in all cases.
⚡ Live Visualizer — Merge Sort
Press ▶ Start to begin
🐍 Python Implementation
def merge_sort(arr):
    # Base case: a single element is already sorted
    if len(arr) <= 1:
        return arr

    # Divide: split at midpoint
    mid = len(arr) // 2
    left  = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])

    # Conquer: merge the sorted halves
    return merge(left, right)


def merge(left, right):
    """Merge two sorted arrays into one sorted array."""
    result = []
    i = j = 0

    # Compare front elements of each half
    while i < len(left) and j < len(right):
        if left[i] <= right[j]:
            result.append(left[i]);  i += 1
        else:
            result.append(right[j]); j += 1

    # Append remaining elements
    result.extend(left[i:])
    result.extend(right[j:])
    return result


# ── Example Usage ──────────────────────────────────────
nums = [38, 27, 43, 3, 9, 82, 10]
result = merge_sort(nums)

print("Original:", nums)    # [38, 27, 43, 3, 9, 82, 10]
print("Sorted:  ", result)  # [3, 9, 10, 27, 38, 43, 82]


# ── In-place version (more memory-efficient) ───────────
def merge_sort_inplace(arr, left=0, right=None):
    if right is None:
        right = len(arr) - 1
    if left < right:
        mid = (left + right) // 2
        merge_sort_inplace(arr, left, mid)
        merge_sort_inplace(arr, mid + 1, right)
        merge_inplace(arr, left, mid, right)

def merge_inplace(arr, left, mid, right):
    L = arr[left:mid+1]
    R = arr[mid+1:right+1]
    i = j = 0; k = left
    while i < len(L) and j < len(R):
        if L[i] <= R[j]: arr[k] = L[i]; i += 1
        else:            arr[k] = R[j]; j += 1
        k += 1
    while i < len(L): arr[k] = L[i]; i += 1; k += 1
    while j < len(R): arr[k] = R[j]; j += 1; k += 1

Insertion Sort

Build Sorted Array One Element at a Time
A simple sorting algorithm that builds the final sorted array one element at a time. It iterates through the array, comparing each element to the ones before it and inserting it into its correct position — much like sorting playing cards in your hand.
⚡ Live Visualizer — Insertion Sort
Press ▶ Start to begin
🐍 Python Implementation
def insertion_sort(arr):
    # Start from index 1 (first element is trivially sorted)
    for i in range(1, len(arr)):
        key = arr[i]  # Element to be inserted
        j = i - 1

        # Shift elements right until we find the correct position
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1

        # Insert key in its correct position
        arr[j + 1] = key

    return arr


# ── Example Usage ──────────────────────────────────────
nums = [12, 11, 13, 5, 6]
result = insertion_sort(nums.copy())

print("Original:", nums)    # [12, 11, 13, 5, 6]
print("Sorted:  ", result)  # [5, 6, 11, 12, 13]


# ── Verbose trace ──────────────────────────────────────
def insertion_sort_verbose(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        print(f"Inserting key={key} at position {i}")
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
        print(f"  Array: {arr}")
    return arr

Beyond The Basics

Other Sorting Algorithms

⚡
Quick Sort

Selects a pivot element, partitions the list around the pivot (elements smaller go left, larger go right), then recursively sorts the two sub-lists. Average time: O(n log n), Worst: O(n²).

🏔
Heap Sort

Creates a binary max-heap from the array, then repeatedly extracts the maximum element and adds it to the sorted portion. Guarantees O(n log n) time in all cases. In-place but not stable.

🔢
Radix Sort

Sorts elements based on their individual digits or characters — from least significant to most significant digit. Achieves linear O(nk) time, where k is the number of digits. Non-comparison based.

🐍 Quick Sort — Python Implementation
def quick_sort(arr):
    if len(arr) <= 1:
        return arr

    pivot = arr[len(arr) // 2]    # Pick middle as pivot
    left   = [x for x in arr if x <  pivot]
    middle = [x for x in arr if x == pivot]
    right  = [x for x in arr if x >  pivot]

    return quick_sort(left) + middle + quick_sort(right)


# In-place Quick Sort with Lomuto partition scheme
def quick_sort_inplace(arr, low=0, high=None):
    if high is None:
        high = len(arr) - 1
    if low < high:
        pi = partition(arr, low, high)
        quick_sort_inplace(arr, low, pi - 1)
        quick_sort_inplace(arr, pi + 1, high)

def partition(arr, low, high):
    pivot = arr[high]
    i = low - 1
    for j in range(low, high):
        if arr[j] <= pivot:
            i += 1
            arr[i], arr[j] = arr[j], arr[i]
    arr[i + 1], arr[high] = arr[high], arr[i + 1]
    return i + 1


# ── Example ────────────────────────────────────────────
nums = [10, 7, 8, 9, 1, 5]
print(quick_sort(nums))              # [1, 5, 7, 8, 9, 10]
quick_sort_inplace(nums); print(nums) # [1, 5, 7, 8, 9, 10]

Python Built-ins

Sorting in Python

list.sort() sorts the list in-place — the original list is modified and None is returned. It uses Python's Timsort algorithm (hybrid of Merge Sort and Insertion Sort).

ParameterRequiredDescription
key Optional A function that takes an element and returns a comparison value. Default compares elements directly.
reverse Optional Boolean. If True, sorts in descending order. Default is False (ascending). 
# ── list.sort() ────────────────────────────────────────
# Signature: list.sort(key=None, reverse=False)

# 1. Basic ascending sort (default)
fruits = ['banana', 'apple', 'cherry', 'date']
fruits.sort()
print(fruits)  # ['apple', 'banana', 'cherry', 'date']

# 2. Descending order with reverse=True
numbers = [3, 1, 4, 1, 5, 9, 2]
numbers.sort(reverse=True)
print(numbers)  # [9, 5, 4, 3, 2, 1, 1]

# 3. Sort by key — string length
words = ['banana', 'fig', 'apple', 'kiwi']
words.sort(key=len)
print(words)  # ['fig', 'kiwi', 'apple', 'banana']

# 4. Sort list of dictionaries by a field
students = [
    {'name': 'Alice', 'grade': 88},
    {'name': 'Bob',   'grade': 95},
    {'name': 'Carol', 'grade': 72},
]
students.sort(key=lambda s: s['grade'], reverse=True)
for s in students:
    print(f"{s['name']}: {s['grade']}")
# Bob: 95 | Alice: 88 | Carol: 72

# 5. Sort by multiple criteria
data = [('Alice', 90), ('Bob', 85), ('Carol', 90), ('Dave', 85)]
data.sort(key=lambda x: (-x[1], x[0]))  # grade desc, name asc
print(data)
# [('Alice', 90), ('Carol', 90), ('Bob', 85), ('Dave', 85)]

# ⚠ NOTE: .sort() returns None — the original list is modified
nums = [3, 1, 2]
result = nums.sort()
print(result)  # None  ← NOT the sorted list!
print(nums)    # [1, 2, 3]  ← original was modified

sorted() returns a new sorted list — the original iterable is not modified. Works on any iterable (lists, tuples, strings, dicts, generators).

ParameterRequiredDescription
iterable Required Any iterable object (list, tuple, string, dict, etc.) that you want to sort.
key Optional A function that takes an element and returns a comparison value. Default compares directly.
reverse Optional Boolean. If True, sorts in descending order. Default is False. 
# ── sorted() ───────────────────────────────────────────
# Signature: sorted(iterable, key=None, reverse=False)

# 1. Returns NEW list — original unchanged
nums = [3, 1, 4, 1, 5, 9, 2]
sorted_nums = sorted(nums)
print("Original:", nums)        # [3, 1, 4, 1, 5, 9, 2]
print("New list:", sorted_nums)  # [1, 1, 2, 3, 4, 5, 9]

# 2. Works on ANY iterable
letters = ('z', 'a', 'm', 'b')       # tuple
print(sorted(letters))              # ['a', 'b', 'm', 'z']

word = "python"                        # string
print(sorted(word))                  # ['h', 'n', 'o', 'p', 't', 'y']

# 3. Reverse sort
print(sorted(nums, reverse=True))  # [9, 5, 4, 3, 2, 1, 1]

# 4. Sort with key function
words = ['banana', 'fig', 'apple', 'date']
by_last_char = sorted(words, key=lambda w: w[-1])
print(by_last_char)  # ['banana', 'apple', 'date', 'fig']

# 5. Sort tuples / complex structures
students = [('Alice', 88), ('Bob', 95), ('Carol', 72)]
by_grade   = sorted(students, key=lambda s: s[1])
by_name    = sorted(students, key=lambda s: s[0])
print("By grade:", by_grade)
# [('Carol', 72), ('Alice', 88), ('Bob', 95)]
print("By name: ", by_name)
# [('Alice', 88), ('Bob', 95), ('Carol', 72)]

# 6. Sort dict keys or values
scores = {'Alice': 88, 'Bob': 95, 'Carol': 72}
print(sorted(scores))                          # sort keys
print(sorted(scores, key=scores.get))          # sort by value
print(sorted(scores.items(), key=lambda x: x[1])) # items by value

# ── When to use which ──────────────────────────────────

# Use .sort() when you want to modify in-place
data = [5, 2, 8, 1]
data.sort()  # data is now [1, 2, 5, 8]

# Use sorted() when you need the original unchanged
data = [5, 2, 8, 1]
new_data = sorted(data)  # data still [5, 2, 8, 1]

# Use sorted() on non-list iterables
gen = (x**2 for x in [3, 1, 4, 1, 5])
print(sorted(gen))  # [1, 1, 9, 16, 25]

# Common: sort and keep both versions
original = ['banana', 'apple', 'cherry']
sorted_v  = sorted(original)
print(original)  # ['banana', 'apple', 'cherry']
print(sorted_v)  # ['apple', 'banana', 'cherry']
